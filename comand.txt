git  config --global user.name "your user name"   [to add your user name]

git config --global user.email  "your email id"    [to add email id]

git init   [to initialize an empty git repository]

git status   [to check untracked files]

git add -a   [to add all untracked files to staging area]

git commit -m "commit message"   [to moved staged files to local repository]

git push   [IR=>RR]

git pull  [RR=>changes=>LWS]

git clone  [RR=>exact copy=>LWS]

git remote add remote_name remote_path   [adding remote]

git push remotename branch  [push]

git branch  [listing all branches]
